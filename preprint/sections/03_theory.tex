\section{Theory}\label{sec:theory}

\subsection{Transport of subgroup-bin reliability}

\begin{proposition}[Transport identity for subgroup-bin reliability]\label{prop:transport}
Under \cref{ass:covshift,ass:overlap}, for every bin \(b\) and subgroup \(g\in\cG\),
\[
\mu_{b,g}^Q
=
\EE_P[\wt(X)A_{b,g}(X)]
\]
and
\[
\theta_{b,g}^Q
=
\EE_P[\wt(X)(Y-\score(X))A_{b,g}(X)].
\]
If \(\mu_{b,g}^Q>0\), then
\[
r_{b,g}^Q
=
\frac{\EE_P[\wt(X)(Y-\score(X))A_{b,g}(X)]}{\EE_P[\wt(X)A_{b,g}(X)]}.
\]
\end{proposition}

The numerator is the weighted source residual mass for one subgroup-bin cell, and the denominator is the weighted source estimate of that cell's target mass.

\subsection{Local ESS is the correct variance scale}

We now study the estimator that matters most for worst-group auditing:
\[
\widehat r_{b,g}
=
\frac{\sum_{i=1}^{n}\wt_i(Y_i-\score_i)A_i}{\sum_{i=1}^{n}\wt_iA_i},
\]
where for short we write \(\wt_i=\wt(X_i)\), \(\score_i=\score(X_i)\), and \(A_i=A_{b,g}(X_i)\).

\begin{assumption}[Bounded conditional residual variance]\label{ass:variance}
For each audited cell \((b,g)\), the conditional residual variance
\[
\Var(Y-\score(X)\given X=x)
\]
is bounded on that cell by a finite constant \(\sigma_{b,g}^2\).
\end{assumption}

\begin{theorem}[Local ESS governs subgroup-bin variance]\label{thm:local-ess}
Assume \cref{ass:covshift,ass:overlap,ass:variance}. Fix a subgroup-bin cell \((b,g)\) and suppose \(\sum_{i=1}^{n}\wt_iA_i>0\). Define the conditional weighted mean residual
\[
\bar r_{b,g}(X_{1:n})
=
\frac{\sum_{i=1}^{n}\wt_iA_i\,\EE[Y_i-\score_i\given X_i]}{\sum_{i=1}^{n}\wt_iA_i}.
\]
Then, conditional on \(X_1,\dots,X_n\),
\[
\EE[\widehat r_{b,g}\given X_{1:n}]
=
\bar r_{b,g}(X_{1:n})
\]
and
\begin{equation}\label{eq:local-variance}
\Var(\widehat r_{b,g}\given X_{1:n})
=
\frac{\sum_{i=1}^{n}\wt_i^2A_i\,\Var(Y_i-\score_i\given X_i)}{\left(\sum_{i=1}^{n}\wt_iA_i\right)^2}
\le
\frac{\sigma_{b,g}^2}{{\ess}_{b,g}}.
\end{equation}
\end{theorem}

Inside one audited cell, \(\widehat r_{b,g}\) is just a weighted average of residuals. The variance of a weighted average is the sum of squared normalized weights times the conditional residual variances, which is why \({\ess}_{b,g}\), not the global \(\ess\), is the right information scale.

\subsection{False worst-group alarms under clean covariate shift}

The previous theorem becomes especially sharp when the model is locally well calibrated. That is exactly the setting used in our new synthetic experiment.

\begin{corollary}[False-alarm bound for worst-group audits]\label{cor:false-alarm}
Assume the conditions of \cref{thm:local-ess}. Suppose that for every audited cell \((b,g)\),
\[
\EE[Y-\score(X)\given X=x]=0
\qquad\text{whenever } A_{b,g}(x)=1.
\]
Then for any threshold \(t>0\),
\begin{equation}\label{eq:false-alarm-bound}
\PP\!\left(
\max_{1\le b\le B,\ g\in\cG}
|\widehat r_{b,g}|>t
\;\middle|\;
X_{1:n}
\right)
\le
\sum_{b=1}^{B}\sum_{g\in\cG}\frac{\sigma_{b,g}^2}{t^2{\ess}_{b,g}}.
\end{equation}
In particular, if
\[
{\ess}_{\min}^{\mathrm{local}}
=
\min_{1\le b\le B,\ g\in\cG}{\ess}_{b,g},
\]
then
\[
\PP\!\left(
\max_{b,g}|\widehat r_{b,g}|>t
\;\middle|\;
X_{1:n}
\right)
\le
\frac{BG\,\sigma_{\max}^2}{t^2{\ess}_{\min}^{\mathrm{local}}},
\]
where \(\sigma_{\max}^2=\max_{b,g}\sigma_{b,g}^2\).
\end{corollary}

Even when there is no true subgroup failure, a worst-group audit can exceed a practical threshold simply because one audited cell has too little weighted information.

\subsection{Why global ESS can be non-diagnostic}

The next result shows that local and global overlap can tell very different stories.

\begin{proposition}[Global ESS can remain large while one audited cell degenerates]\label{prop:global-misleading}
Fix any target cell mass \(\pi\in(0,1)\) and any integer \(N\ge 2\). There exists a weighted sample \((A_i,\wt_i)_{i=1}^{N}\) such that
\[
\frac{1}{N}\sum_{i=1}^{N}\wt_iA_i=\pi,
\qquad
{\ess}_{\mathrm{cell}}=1,
\]
but the global effective sample size satisfies
\begin{equation}\label{eq:global-counterexample}
\ess
=
\frac{1}{\pi^2 + (1-\pi)^2/(N-1)}.
\end{equation}
\end{proposition}

\begin{proof}[Construction]
Take exactly one active point, say \(A_1=1\), and let \(A_2=\cdots=A_N=0\). Assign weight \(\wt_1=N\pi\) to the active point and equal weights
\[
\wt_2=\cdots=\wt_N=\frac{N(1-\pi)}{N-1}
\]
to the remaining \(N-1\) points. Then the weighted cell mass is \(\pi\), the active cell contains only one weighted point, and therefore \({\ess}_{\mathrm{cell}}=1\). The global ESS is obtained by direct substitution into \cref{eq:global-ess-def}, which yields \cref{eq:global-counterexample}.
\end{proof}

For example, if \(\pi=0.05\), then the right-hand side of \cref{eq:global-counterexample} is close to \(400\) for large \(N\). So a reassuring global ESS number does not prevent one audited cell from being statistically degenerate.

\subsection{Real hidden failures still matter}

The local-ESS story does not replace the hidden-failure story. It complements it.

\begin{proposition}[Aggregate calibration can hide genuine subgroup failure]\label{prop:hidden}
There exist distributions \(P,Q\), a score function \(\score\), and a subgroup family \(\cG\) such that
\[
\Cal_{Q,\Bin}(\ind)=0
\qquad\text{but}\qquad
\WCE_{Q,\Bin}(\cG)>0.
\]
\end{proposition}

\begin{proof}[Construction]
Take two groups \(g_A\) and \(g_B\) with equal target mass. Let \(Y=0\) almost surely on group \(A\) and \(Y=1\) almost surely on group \(B\). Let the score be constant, \(\score(X)=1/2\), and use a single score bin. Then the aggregate residual is
\[
\EE_Q[Y-\score(X)]
=
\frac{1}{2}(0-1/2)+\frac{1}{2}(1-1/2)=0,
\]
so aggregate calibration is perfect. But inside each subgroup the absolute calibration gap is \(1/2\), so subgroup reliability is poor.
\end{proof}

\Cref{prop:hidden} explains why worst-group auditing is needed in the first place. \Cref{cor:false-alarm} explains why that audit must also be accompanied by local support diagnostics.

\subsection{Estimated weights and selective deployment}

The local-ESS theorem isolates the sampling-noise part of the problem under oracle weights. With estimated weights \(\widehat w\), the total error decomposes as
\[
|\widehat r_{b,g}(\widehat w)-r_{b,g}^Q|
\le
|\widehat r_{b,g}(\widehat w)-\widehat r_{b,g}(w)|
+
|\widehat r_{b,g}(w)-r_{b,g}^Q|.
\]
The first term is weight-estimation error. The second is the local-overlap term analyzed above. The same logic also applies to selective deployment. If \(\cA_\tau=\{x:u(x)\ge\tau\}\) is the accepted set at threshold \(\tau\), then the accepted-set subgroup indicator is
\[
\widetilde A_{g,\tau}(x)=\ind\{x\in\cA_\tau\}g(x).
\]
Accepted-set risk, accepted-set calibration, and subgroup abstention gaps are therefore governed by the same local-overlap mechanism \citep{Geifman2017Selective, Jones2020SelectiveDisparities}.
